function fizzBuzz(n) { let res = []; for (let i = 1; i <= n; i++) { if (i % 3 === 0 && i % 5 === 0) { res.push("FizzBuzz"); } else if (i % 3 === 0) { res.push("Fizz"); } else if (i % 5 === 0) { res.push("Buzz"); // Fixed the typo here } else { res.push(String(i)); } } return res; } // Example usage: console.log(fizzBuzz(15)); // Expected output: ["1", "2", "Fizz", "4", "Buzz", "Fizz", "7", "8", "Fizz", "Buzz", "11", "Fizz", "13", "14", "FizzBuzz"]
