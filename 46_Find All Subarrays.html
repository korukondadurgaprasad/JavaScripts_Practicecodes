def findAllSubarrays(arr): subarrays = [] # Iterate through the array to find all subarrays for start in range(len(arr)): for end in range(start, len(arr)): subarrays.append(arr[start:end+1]) # Add the subarray from index start to end return subarrays # Example usage: arr = [1, 2, 3] result = findAllSubarrays(arr) print(result) # Output: [[1], [1, 2], [1, 2, 3], [2], [2, 3], [3]]
