def find_consonants(arr): vowels = "aeiouAEIOU" consonants = [] for word in arr: for char in word: if char.isalpha() and char not in vowels: consonants.append(char) return consonants # Example usage: arr = ["hello", "world", "Python", "array"] print(find_consonants(arr)) # Output: ['h', 'l', 'l', 'w', 'r', 'l', 'd', 'P', 'y', 't', 'h', 'n', 'r', 'r']
