import math def is_perfect_square(n): # If n is negative, it cannot be a perfect square if n < 0: return False # Calculate the square root of n and round it to the nearest integer root = int(math.sqrt(n)) # Check if the square of the rounded root equals the original number return root * root == n # Input number = int(input().strip()) # Output if is_perfect_square(number): print("Yes") else: print("No")
